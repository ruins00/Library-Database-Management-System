CREATE TRIGGER STUI
AFTER INSERT ON STUDENT
FOR EACH ROW
UPDATE TOTAL_COUNT SET TOTAL = TOTAL + 1 WHERE ENTITY = 'STUDENT';
END;

CREATE TRIGGER BUI
AFTER INSERT ON BOOK
FOR EACH ROW
UPDATE TOTAL_COUNT SET TOTAL = TOTAL + 1 WHERE ENTITY = 'BOOK';
END;

CREATE TRIGGER BUD
AFTER DELETE ON BOOK
FOR EACH ROW
UPDATE TOTAL_COUNT SET TOTAL = TOTAL - 1 WHERE ENTITY = 'BOOK';
END;


create table TOTAL_COUNT(ENTITY VARCHAR(20),TOTAL INTEGER);


CREATE TRIGGER SUD
AFTER DELETE ON STUDENT
FOR EACH ROW
UPDATE TOTAL_COUNT SET TOTAL = TOTAL - 1 WHERE ENTITY = 'STUDENT';
END;
